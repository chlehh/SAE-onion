CREATE TABLE routeurs (
    id INT AUTO_INCREMENT PRIMARY KEY,   -- Identifiant unique pour chaque routeur
    nom VARCHAR(50),                     -- Nom du routeur
    adresse_ip VARCHAR(50),              -- Adresse IP du routeur
    type VARCHAR(50),                    -- Type de routeur (par exemple, routeur ou client)
    next_hop VARCHAR(50),                -- Prochain saut (next hop) du routeur
    port INT(11),                        -- Port d'écoute du routeur
    cle_publique TEXT                    -- Clé publique du routeur
);
CREATE TABLE logs (
    id INT AUTO_INCREMENT PRIMARY KEY,    -- Identifiant unique pour chaque log
    message_id VARCHAR(50),               -- Identifiant du message
    routeur VARCHAR(50),                  -- Nom du routeur qui a traité le message
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP  -- Date et heure de l'enregistrement du log
);
